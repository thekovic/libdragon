737inline void rdpq_set_tile(rdpq_tile_t tile,
738 tex_format_t format,
739 int32_t tmem_addr,
740 uint16_t tmem_pitch,
741 const rdpq_tileparms_t *parms)
742{
743 static const rdpq_tileparms_t default_parms = {0};
744 if (!parms) parms = &default_parms;
745 else {
746 assertf(parms->s.shift >= -5 && parms->s.shift <= 10, "invalid s shift %d: must be in [-5..10]", parms->s.shift);
747 assertf(parms->t.shift >= -5 && parms->t.shift <= 10, "invalid t shift %d: must be in [-5..10]", parms->t.shift);
748 }
749 bool fixup = false;
750 bool reuse = false;
751 uint32_t cmd_id = RDPQ_CMD_SET_TILE;
752 if (tmem_addr & (RDPQ_AUTOTMEM | RDPQ_AUTOTMEM_REUSE(0))) {
753 cmd_id = RDPQ_CMD_AUTOTMEM_SET_TILE;
754 reuse = (tmem_addr & RDPQ_AUTOTMEM_REUSE(0)) != 0;
755 fixup = true;
756 tmem_addr &= ~(RDPQ_AUTOTMEM | RDPQ_AUTOTMEM_REUSE(0));
757 } else {
758 assertf((tmem_addr % 8) == 0, "invalid tmem_addr %ld: must be multiple of 8", tmem_addr);
759 tmem_addr /= 8;
760 }
761 assertf((tmem_pitch % 8) == 0, "invalid tmem_pitch %d: must be multiple of 8", tmem_pitch);
762 extern void __rdpq_write8_syncchange(uint32_t, uint32_t, uint32_t, uint32_t);
763 extern void __rdpq_fixup_write8_syncchange(uint32_t, uint32_t, uint32_t, uint32_t);
764 (fixup ? __rdpq_fixup_write8_syncchange : __rdpq_write8_syncchange)(cmd_id,
765 _carg(format, 0x1F, 19) | _carg(reuse, 0x1, 18) | _carg(tmem_pitch/8, 0x1FF, 9) | _carg(tmem_addr, 0x1FF, 0),
766 _carg(tile, 0x7, 24) | _carg(parms->palette, 0xF, 20) |
767 _carg(parms->t.clamp | (parms->t.mask == 0), 0x1, 19) | _carg(parms->t.mirror, 0x1, 18) | _carg(parms->t.mask, 0xF, 14) | _carg(parms->t.shift, 0xF, 10) |
768 _carg(parms->s.clamp | (parms->s.mask == 0), 0x1, 9) | _carg(parms->s.mirror, 0x1, 8) | _carg(parms->s.mask, 0xF, 4) | _carg(parms->s.shift, 0xF, 0),
769 AUTOSYNC_TILE(tile));
770}
